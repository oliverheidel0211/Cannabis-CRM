    function restoreSession() {

        try {

            const session = JSON.parse(sessionStorage.getItem(CONFIG.SESSION_KEY));

            if (session && session.timestamp && (Date.now() - session.timestamp) < 3600000) {

    }

    async function handleLogin() {

        if (Date.now() < loginLockedUntil) {

            const remaining = Math.ceil((loginLockedUntil - Date.now()) / 60000);

            showToast(`Zu viele Versuche. Bitte warte ${remaining} Minuten.`, "error");

            return;

        }


        const email = document.getElementById("loginEmail").value.trim().toLowerCase();

        const pass = document.getElementById("loginPassword").value;

        const emailInput = document.getElementById("loginEmail");

        const passInput = document.getElementById("loginPassword");


        let hasError = false;

        if (!email || !email.includes('@')) { emailInput.classList.add('error'); hasError = true; }

        else { emailInput.classList.remove('error'); }

        if (!pass) { passInput.classList.add('error'); hasError = true; }

        else { passInput.classList.remove('error'); }

        if (hasError) return;


        const user = USERS[email];

        if (user) {

            const isValid = user.passHash === pass;

            if (isValid) {

                loginAttempts = 0;

                currentUser = { name: user.name, role: user.role };

                saveSession();

                activateAppLayer();

                showToast(`Willkommen zurück, ${user.name}!`, "success");

                return;

            }

        }


        loginAttempts++;

        if (loginAttempts >= CONFIG.MAX_LOGIN_ATTEMPTS) {

            loginLockedUntil = Date.now() + CONFIG.LOGIN_COOLDOWN_MS;

            showToast(`Zu viele Fehlversuche. Account für 5 Minuten gesperrt.`, "error");

        } else {

            showToast(`Zugangsdaten ungültig. Noch ${CONFIG.MAX_LOGIN_ATTEMPTS - loginAttempts} Versuche.`, "error");

        }

        passInput.classList.add('error');

    }

    function getWiedervorlageStatus(dateStr) {

        if (!dateStr) return { status: 'none', label: '—' };

        const today = new Date(); today.setHours(0, 0, 0, 0);

        const wvDate = new Date(dateStr); wvDate.setHours(0, 0, 0, 0);

        const diffDays = Math.floor((wvDate - today) / (1000 * 60 * 60 * 24));

        if (diffDays < 0) return { status: 'overdue', label: `⚠️ Überfällig (${Math.abs(diffDays)} Tage)`, class: 'overdue' };

        if (diffDays === 0) return { status: 'today', label: '🔥 Heute fällig', class: 'today' };

        if (diffDays === 1) return { status: 'tomorrow', label: '⏰ Morgen', class: '' };

        return { status: 'future', label: `📅 ${dateStr}`, class: '' };

    }


    function getSortFunction() {

        const sortValue = document.getElementById('sortSelect').value;

        switch(sortValue) {

            case 'prio-desc': return (a, b) => (a.prioritaet || 'D').localeCompare(b.prioritaet || 'D');

            case 'prio-asc': return (a, b) => (b.prioritaet || 'D').localeCompare(a.prioritaet || 'D');

            case 'umsatz-desc': return (a, b) => (b.gesamtUmsatzLive || 0) - (a.gesamtUmsatzLive || 0);

            case 'umsatz-asc': return (a, b) => (a.gesamtUmsatzLive || 0) - (b.gesamtUmsatzLive || 0);

            case 'wiedervorlage': return (a, b) => {

                const da = a.wiedervorlage ? new Date(a.wiedervorlage) : new Date('2099-01-01');

                const db = b.wiedervorlage ? new Date(b.wiedervorlage) : new Date('2099-01-01');

                return da - db;

            };

            case 'name': return (a, b) => (a.apothekenname || '').localeCompare(b.apothekenname || '');

            default: return () => 0;

        }

    }

    function escapeHtml(str) { if(str === null || str === undefined) return ""; return String(str).replace(/[&<>"']/g, m => ({ '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#39;' }[m])); }


    function debounce(func, wait) {

        return function executedFunction(...args) {

            const later = () => { clearTimeout(searchDebounceTimer); func(...args); };

            clearTimeout(searchDebounceTimer);

            searchDebounceTimer = setTimeout(later, wait);

        };

    }
